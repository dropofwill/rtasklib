def get_version
  raw, ec = Execute.task(@create_new, "rc.data.location=#{data_location}",
                                   "_version")
  if ec == 0
    return to_gem_version(raw)
  else
    return nil
  end
end

def to_gem_version raw
  std_ver = raw.chomp.gsub(' ','.').delete('(').delete(')')
  p std_ver
  Gem::Version.new std_ver
end
