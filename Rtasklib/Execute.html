def each_popen3 program='task', *opts, &block
  popen3(program, *opts) do |i, o, e, t|
    o.each_line do |l|
      yield(l, i, o, e, t)
    end
  end
end

def handle_response stderr, thread
  unless thread.value.success?
    puts stderr.read
    raise thread.inspect
  end
end

- (Object) popen3(program = 'task', *opts, &block)

Use Open3#popen3 to execute a unix program with an array of options and an optional block to handle the response. Passes: STDIN, STDOUT, STDERR, and the thread to that block.

For example:

Execute.popen3(“task”, “export”) do |i, o, e, t|

# Arbitrary code to handle the response...

end

Parameters:

  • program (String) (defaults to: 'task')
  • opts (Array<String>) —

    args to pass directly to the program

  • block (Block) —

    to execute after thread is successful

def popen3 program='task', *opts, &block
  execute = opts.unshift(program)
  execute = execute.join(" ")
  warn execute

  Open3.popen3(execute) do |i, o, e, t|
    handle_response(e, t)
    yield(i, o, e, t) if block_given?
  end
end
